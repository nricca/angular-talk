Service

  • Singleton
  • Injéctable
  • Services prédéfinis par Angular ($http, $route, $log, etc.)

Exemple

<input type="text" ng-model="myHero">
<button class="btn btn-primary" ng-click="addHero(myHero)">
    Ajouter
</button>
<ul>
    <li ng-repeat="hero in heroes">{{ hero }}</li>
</ul>
                
app.controller("MyCtrl", function($scope, heroService) {

    $scope.heroes = heroService.heroes();

    $scope.addHero = function (hero) {
        heroService.add(hero);
    }
});
                
angular.module("app").factory("heroService", function() {
    var heroes = new Array("Batman", "Superman");
    return {
        add : function(hero) {
            heroes.push(hero);
        },
        heroes : function() {
            return heroes;
        }}
});
