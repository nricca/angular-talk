Routeur

  • Utilisation via ng-view
  • Associe du contenu à une url (deep linking)

Exemple

<body ng-app="app">
    <div ng-view></div>
</body>
                
angular.module('app')
    .config(function ($routeProvider) {
        $routeProvider
            .when('/', {
                templateUrl: 'partials/presentation.html',
                controller: 'MyCtrl'
            })
            .when('/page-de-test', {
                templateUrl: 'partials/pageDeTest.html',
                controller: 'MyCtrl'
            });
    });
                
    Aller à la page de test
